CREATE TABLE `caijiqi` (
  `id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '自增主键',
  `uid` varchar(40) DEFAULT NULL COMMENT '去重id',
  `uuid` varchar(65) DEFAULT NULL COMMENT '备用字段',
  `title` text COMMENT '标题',
  `link` text COMMENT '链接',
  `intro` text COMMENT 'json',
  `abs` varchar(10) DEFAULT NULL COMMENT 'html/josn/采集器',
  `content` longtext COMMENT '正文内容',
  `publish_time` varchar(20) DEFAULT NULL COMMENT '发布时间',
  `purchaser` varchar(255) DEFAULT NULL COMMENT '招标单位',
  `proxy` varchar(255) DEFAULT NULL COMMENT '代理人',
  `create_time` varchar(20) DEFAULT NULL COMMENT '入库时间',
  `update_time` varchar(20) DEFAULT NULL COMMENT '更新时间',
  `deleted` varchar(10) DEFAULT NULL COMMENT '是否删除',
  `province` varchar(80) DEFAULT NULL COMMENT '省市区',
  `base` varchar(255) DEFAULT NULL COMMENT '备份字段',
  `type` varchar(20) DEFAULT NULL COMMENT '类型',
  `items` varchar(20) DEFAULT NULL COMMENT '行业',
  `data_source` varchar(20) DEFAULT NULL COMMENT '网站编码',
  `end_time` varchar(20) DEFAULT NULL COMMENT '结束时间',
  `status` varchar(30) DEFAULT NULL COMMENT '招标状态',
  `serial` varchar(255) DEFAULT NULL COMMENT '采购编号',
  PRIMARY KEY (`id`) USING BTREE,
  UNIQUE KEY `id_index` (`id`) USING BTREE,
  UNIQUE KEY `uid_index` (`uid`) USING BTREE,
  KEY `pubu` (`publish_time`) USING BTREE,
  KEY `crawler_time` (`create_time`) USING BTREE,
  FULLTEXT KEY `title` (`title`),
  FULLTEXT KEY `link` (`link`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8 ROW_FORMAT=DYNAMIC;
